let add_local env name content = match env with
        | [] -> [[(name, content)]]
        | h::t -> 
                let rec aux env name content = match env with
                        []                      -> [(name, content)]
                        | (n, c)::t when name = n -> (name, content)::t
                        | h::t                    -> h::(aux t name content)
                in (aux h name content)::t
